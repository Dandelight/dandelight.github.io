 91 func Initcpp()
 92   let l = 0
 93   let l = l + 1 | call setline(l,'#include<bits/stdc++.h>')
 94   let l = l + 1 | call setline(l,'using namespace std;')
 95   let l = l + 1 | call setline(l,'int solve() {')
 96   let l = l + 1 | call setline(l,' ')
 97   let l = l + 1 | call setline(l,'  return 0;')
 98   let l = l + 1 | call setline(l,'}')
 99   let l = l + 1 | call setline(l,'int main() {')
100   let l = l + 1 | call setline(l,'  ios_base::sync_with_stdio(0);')
101   let l = l + 1 | call setline(l,'  cin.tie(0); cout.tie(0);')
102   let l = l + 1 | call setline(l,'  // freopen("in.txt","r",stdin);')
103   let l = l + 1 | call setline(l,'  // freopen("out.txt","w",stdout);')
104   let l = l + 1 | call setline(l,'  int tc; tc = 1;')
105   let l = l + 1 | call setline(l,'  for(int t = 1; t <= tc; t++) {')
106   let l = l + 1 | call setline(l,'    // cout << "Case #" << t << ": "')
107   let l = l + 1 | call setline(l,'    solve();')
108   let l = l + 1 | call setline(l,'  }')
109   let l = l + 1 | call setline(l,'  return 0;')
110   let l = l + 1 | call setline(l,'}')
111   exec "4"
112 endfunc
